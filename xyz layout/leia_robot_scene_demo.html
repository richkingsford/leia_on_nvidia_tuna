const rightStack = { x: 4.0, y: 2.2 };
const frontStack = { x: 2.15, y: 3.25 };
const view = LeiaRobotScene.createLeiaRobotScene('#scene');

view.addBrickToStack(rightStack);
view.removeTopBrick(frontStack);
view.stickTopBrick(rightStack);
view.moveRobotForward(0.24);
view.unstickBrick();
